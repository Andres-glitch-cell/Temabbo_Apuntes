    <script>
        // Animaci√≥n entrada del t√≠tulo principal letra por letra
        const mainTitle = document.querySelector('h1');
        mainTitle.innerHTML = mainTitle.textContent.replace(/\S/g, "<span class='letter'>$&</span>");

        anime.timeline({ loop: false })
            .add({
                targets: '.letter',
                opacity: [0, 1],
                translateY: [20, 0],
                easing: "easeOutExpo",
                duration: 850,
                delay: (el, i) => 50 * i
            });

        // Animaci√≥n de aparici√≥n de secciones al hacer scroll
        document.querySelectorAll('.section').forEach(section => {
            section.style.opacity = 0;
            anime({
                targets: section,
                opacity: [0, 1],
                translateY: [50, 0],
                delay: anime.stagger(100, { start: 400 }),
                duration: 1000,
                easing: 'easeOutQuad',
                autoplay: false
            });
        });

        // Animaci√≥n de botones al pasar el mouse
        const animatedButtons = document.querySelectorAll('button, .btn, .scrollBtn');
        animatedButtons.forEach(btn => {
            btn.addEventListener('mouseenter', () => {
                anime({
                    targets: btn,
                    scale: 1.1,
                    duration: 300,
                    easing: 'easeOutExpo'
                });
            });
            btn.addEventListener('mouseleave', () => {
                anime({
                    targets: btn,
                    scale: 1,
                    duration: 300,
                    easing: 'easeOutExpo'
                });
            });
        });

        // Animaci√≥n de √≠conos decorativos rotando
        const decorativeIcons = document.querySelectorAll('.icon-rotating');
        anime({
            targets: decorativeIcons,
            rotate: '1turn',
            duration: 8000,
            loop: true,
            easing: 'linear'
        });

        // Puedes animar elementos con clase .fade-in cuando entran en viewport (extra suave)
        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    anime({
                        targets: entry.target,
                        opacity: [0, 1],
                        translateY: [20, 0],
                        duration: 1000,
                        easing: 'easeOutSine'
                    });
                    observer.unobserve(entry.target);
                }
            });
        }, { threshold: 0.2 });

        document.querySelectorAll('.fade-in').forEach(el => observer.observe(el));
    </script>
